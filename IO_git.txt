#include <iostream>
using namespace std;

//komentarz do usunięcia

int main() {
    // Deklaracja zmiennych
    double liczba1, liczba2, suma;

    // Pobieranie danych od użytkownika
    cout << "Podaj pierwsza liczbe: ";
    cin >> liczba1;

    cout << "Podaj druga liczbe: ";
    cin >> liczba2;

    // Dodawanie liczb
    suma = liczba1 + liczba2;

    // Wyświetlenie wyniku
    cout << "Suma: " << suma << endl;

    return 0;
}
